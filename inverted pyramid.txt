// inverted full pyramid
#include <stdio.h>
int main()
{
   int i,j,n;
   printf("enter height of pyramid ");
   scanf("%d",&n);
   printf("height of pyramid is %d\n",n);
   for(i=n;i>=1;i--)
   {
       for(j=1;j<=2*n-1;j++)
       {
           if(j>=n-(i-1) && j<=n+(i-1))
           {
           printf("*");
           
           }
           else
           printf(" ");
       }
       printf("\n");
   }
    return 0;
}
